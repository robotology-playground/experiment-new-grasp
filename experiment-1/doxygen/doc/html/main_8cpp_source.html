771  void pointFromName()
772  {
773  Bottle cmd,reply;
774  blob_points.clear();
775  contour.clear();
776  cmd.addVocab(Vocab::encode("ask"));
777  Bottle &content=cmd.addList();
778  Bottle &cond_1=content.addList();
779  cond_1.addString("entity");
780  cond_1.addString("==");
781  cond_1.addString("object");
782  content.addString("&&");
783  Bottle &cond_2=content.addList();
784  cond_2.addString("name");
785  cond_2.addString("==");
786  cond_2.addString(objname);
787 
788  portOPCrpc.write(cmd,reply);
789  if(reply.size()>1)
790  {
791  if(reply.get(0).asVocab()==Vocab::encode("ack"))
792  {
793  if (Bottle *b=reply.get(1).asList())
794  {
795  if (Bottle *b1=b->get(1).asList())
796  {
797  cmd.clear();
798  int id=b1->get(0).asInt();
799  cmd.addVocab(Vocab::encode("get"));
800  Bottle &info=cmd.addList();
801  Bottle &info2=info.addList();
802  info2.addString("id");
803  info2.addInt(id);
804  Bottle &info3=info.addList();
805  info3.addString("propSet");
806  Bottle &info4=info3.addList();
807  info4.addString("position_2d_left");
808  }
809  else
810  {
811  yError("no object id provided by OPC!");
812  contour.clear();
813  }
814  }
815  else
816  {
817  yError("uncorrect reply from OPC!");
818  contour.clear();
819  }
820 
821  Bottle reply;
822  if (portOPCrpc.write(cmd,reply))
823  {
824 
825  if (reply.size()>1)
826  {
827  if (reply.get(0).asVocab()==Vocab::encode("ack"))
828  {
829  if (Bottle *b=reply.get(1).asList())
830  {
831  if (Bottle *b1=b->find("position_2d_left").asList())
832  {
833  cv::Point p1,p2,p;
834  p1.x=b1->get(0).asInt();
835  p1.y=b1->get(1).asInt();
836  p2.x=b1->get(2).asInt();
837  p2.y=b1->get(3).asInt();
838  p.x=p1.x+(p2.x-p1.x)/2;
839  p.y=p1.y+(p2.y-p1.y)/2;
840  contour.push_back(p);
841  }
842  else
843  {
844  yError("position_2d_left field not found in the OPC reply!");
845  contour.clear();
846  }
847  }
848  else
849  {
850  yError("uncorrect reply structure received!");
851  contour.clear();
852  }
853  }
854  else
855  {
856  yError("Failure in reply for object 2D point!");
857  contour.clear();
858  }
859  }
860  else
861  {
862  yError("reply size for 2D point less than 1!");
863  contour.clear();
864  }
865  }
866  else
867  yError("no reply from second OPC query!");
868  }
869  else
870  {
871  yError("Failure in reply for object id!");
872  contour.clear();
873  }
874  }
875  else
876  {
877  yError("reply size for object id less than 1!");
878  contour.clear();
879  }

886  /***********************************************************************/
887  deque<Vector> get3Dpoints(ImageOf<PixelRgb> *ImgIn)
888  {
889  Bottle cmd,reply;
890  cmd.addString("Points");
891  int count_blob=0;
892 
893  deque<Vector> p;
894 
895  p.clear();
896 
897  for (size_t i=0; i<blob_points.size(); i++)
898  {
899  cv::Point single_point=blob_points[i];
900  cmd.addInt(single_point.x);
901  cmd.addInt(single_point.y);
902  }
903 
904  if (portSFMRpc.write(cmd,reply))
905  {
906  count_blob=0;
907 
908  for (int idx=0;idx<reply.size();idx+=3)
909  {
910  Vector point(6,0.0);
911 
912  point[0]=reply.get(idx+0).asDouble();
913  point[1]=reply.get(idx+1).asDouble();
914  point[2]=reply.get(idx+2).asDouble();
915 
916  if (ImgIn!=NULL && (filtered==false))
917  {
918  PixelRgb px=ImgIn->pixel(blob_points[count_blob].x,blob_points[count_blob].y);
919  point[3]=px.r;
920  point[4]=px.g;
921  point[5]=px.b;
922  }
923  //else
924  // yInfo()<<"No img received yet!";
925 
926  count_blob+=1;
927 
928  if ((norm(point)>0))
929  {
930  p.push_back(point);
931  }
932  }
933 
934  if (points.size()<=0)
935  {
936  yError("[SuperqComputation]: Some problems in point acquisition!");
937  }
938  }
939  else
940  {
941  yError("[SuperqComputation]: SFM reply is fail!");
942  p.clear();
943  }
944 
945  return p;
946  }
